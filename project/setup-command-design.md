# YellowBull `setup` 命令 — 设计规划

## 1. CLI 入口设计

```python
@click.command()
@click.option("--non-interactive", is_flag=True, help="非交互模式，必须提供所有参数")
@click.option("--provider", type=str, default=None, help="LLM 提供商 (openai/anthropic/ollama)")
@click.option("--model", type=str, default=None, help="Model 名称")
@click.option("--api-key", type=str, default=None, help="API Key")
@click.option("--base-url", type=str, default=None, help="Base URL（代理地址）")
@click.option("--db-path", type=str, default="./data/yellowbull.db", help="数据库路径")
@click.option("--tools-allowed", type=str, default="file,shell,code", help="允许的工具列表")
@click.option("--init-data-only", is_flag=True, help="仅初始化数据目录和数据库")
@click.option("--show-config", is_flag=True, help="显示当前配置后退出")
@click.option("--force", is_flag=True, help="强制覆盖已有 .env 文件")
def setup(
    non_interactive: bool,
    provider: str | None,
    model: str | None,
    api_key: str | None,
    base_url: str | None,
    db_path: str,
    tools_allowed: str,
    init_data_only: bool,
    show_config: bool,
    force: bool,
):
    """初始化 YellowBull 配置、数据目录和数据库。"""
```

## 2. .env 生成逻辑

### 2.1 文件路径

- **默认**: `{project_root}/.env`
- **检测顺序**: `.env` → `pydantic-settings` 自动加载环境变量前缀 `YELLOWBULL_`

### 2.2 内容格式

```ini
# YellowBull Configuration
# Generated by `yellowbull setup`

# === LLM ===
YELLOWBULL_LLM_PROVIDER=openai
YELLOWBULL_LLM_MODEL=gpt-4o
YELLOWBULL_LLM_API_KEY=sk-proj-xxx
YELLOWBULL_LLM_BASE_URL=https://api.openai.com/v1

# === Execution ===
YELLOWBULL_EXEC_STEP_TIMEOUT=120
YELLOWBULL_EXEC_TASK_TIMEOUT=1800
YELLOWBULL_EXEC_MAX_STEPS=100
YELLOWBULL_EXEC_RETRY_LIMIT=2

# === Tools ===
YELLOWBULL_TOOLS_ALLOWED=file,shell,code
YELLOWBULL_SHELL_SAFE_MODE=true

# === Database ===
YELLOWBULL_DATABASE_PATH=./data/yellowbull.db

# === Experience ===
YELLOWBULL_EXPERIENCE_ENABLED=true
```

### 2.3 覆盖保护

| 场景 | 行为 |
|------|------|
| `.env` 不存在 | 直接创建 |
| `.env` 存在 + 无 `--force` | 提示确认（交互模式）/ 报错退出（非交互模式） |
| `.env` 存在 + `--force` | 直接覆盖 |

## 3. 数据目录初始化

### 3.1 目录结构

```
./data/
├── yellowbull.db          # SQLite 主库 (WAL 模式)
└── yellowbull.db-wal      # WAL journal（自动创建）
```

### 3.2 数据库初始化流程

```python
async def _init_database(db_path: Path):
    """同步包装异步数据库初始化"""
    settings = DatabaseSettings(path=str(db_path))
    manager = DatabaseManager(settings=settings)
    await manager.initialize()  # WAL + CREATE TABLE IF NOT EXISTS
    await manager.close()
```

### 3.3 调用方式（CLI 中）

```python
import asyncio

asyncio.run(_init_database(Path(db_path)))
```

## 4. Settings 新增方法

### 4.1 `export_env(path: Path) -> None`

```python
class Settings(BaseSettings):
    # ... 现有字段 ...

    def export_env(self, path: Path) -> None:
        """将当前配置导出为 .env 文件"""
        lines = [
            "# YellowBull Configuration",
            "# Generated by `yellowbull setup`",
            "",
            "# === LLM ===",
            f"YELLOWBULL_LLM_PROVIDER={self.llm.provider}",
            f"YELLOWBULL_LLM_MODEL={self.llm.model}",
            f"YELLOWBULL_LLM_API_KEY={self.llm.api_key}",
            f"YELLOWBULL_LLM_BASE_URL={self.llm.base_url or ''}",
            "",
            "# === Execution ===",
            f"YELLOWBULL_EXEC_STEP_TIMEOUT={self.exec.step_timeout}",
            f"YELLOWBULL_EXEC_TASK_TIMEOUT={self.exec.task_timeout}",
            f"YELLOWBULL_EXEC_MAX_STEPS={self.exec.max_steps}",
            f"YELLOWBULL_EXEC_RETRY_LIMIT={self.exec.retry_limit}",
            "",
            "# === Tools ===",
            f"YELLOWBULL_TOOLS_ALLOWED={self.tools_allowed}",
            f"YELLOWBULL_SHELL_SAFE_MODE={'true' if self.shell_safe_mode else 'false'}",
            "",
            "# === Database ===",
            f"YELLOWBULL_DATABASE_PATH={self.database.path}",
            "",
            "# === Experience ===",
            f"YELLOWBULL_EXPERIENCE_ENABLED={'true' if self.experience_enabled else 'false'}",
        ]
        path.write_text("\n".join(lines), encoding="utf-8")
```

### 4.2 `from_dict(data: dict) -> Settings`

```python
@classmethod
def from_dict(cls, data: dict) -> Settings:
    """从 dict 创建 Settings（用于 CLI 参数 → Settings）"""
    llm_data = {
        "provider": data.get("llm_provider", cls.model_fields["llm"].default.provider),
        "model": data.get("llm_model", cls.model_fields["llm"].default.model),
        "api_key": data.get("llm_api_key", ""),
        "base_url": data.get("llm_base_url"),
    }
    exec_data = {
        "step_timeout": data.get("step_timeout", 120),
        "task_timeout": data.get("task_timeout", 1800),
        "max_steps": data.get("max_steps", 100),
        "retry_limit": data.get("retry_limit", 2),
    }
    db_data = {
        "path": data.get("db_path", "./data/yellowbull.db"),
    }
    return cls(
        llm=LLMSettings(**llm_data),
        exec_=ExecutionSettings(**exec_data),
        database=DatabaseSettings(**db_data),
        tools_allowed=data.get("tools_allowed", "file,shell,code"),
        shell_safe_mode=data.get("shell_safe_mode", True),
        experience_enabled=data.get("experience_enabled", True),
    )
```

## 5. 主流程伪代码

```python
def setup(...):
    if show_config:
        settings = Settings()
        _print_summary(settings)
        return

    if init_data_only:
        _init_data_dirs(db_path)
        asyncio.run(_init_database(Path(db_path)))
        print("[✓] 数据初始化完成")
        return

    if non_interactive:
        # 参数校验
        if not all([provider, model, api_key]):
            raise ClickException("--non-interactive 需要 --provider, --model, --api-key")

        config = {
            "llm_provider": provider,
            "llm_model": model,
            "llm_api_key": api_key,
            "llm_base_url": base_url or (_OLLAMA_DEFAULT_URL if provider == "ollama" else None),
            "db_path": db_path,
            "tools_allowed": tools_allowed,
        }

        env_path = Path(".env")
        if env_path.exists() and not force:
            raise ClickException(f"{env_path} 已存在，使用 --force 覆盖")

        settings = Settings.from_dict(config)
        settings.export_env(env_path)
        _gitignore_add_env(Path(".gitignore"))

        _init_data_dirs(db_path)
        asyncio.run(_init_database(Path(db_path)))

        print("[✓] 初始化完成！配置已写入 .env")
    else:
        # 交互模式
        config = _interactive_wizard()
        env_path = Path(".env")
        if env_path.exists():
            confirm(f"{env_path} 已存在，是否覆盖？", default=False)

        settings = Settings.from_dict(config)
        settings.export_env(env_path)
        _gitignore_add_env(Path(".gitignore"))

        _init_data_dirs(db_path)
        asyncio.run(_init_database(Path(db_path)))

        _print_summary(settings)

        if confirm("测试 LLM 连接？", default=True):
            _test_llm_connection(settings)
```

## 6. 函数职责矩阵

| 函数 | 职责 | 输入 | 输出/副作用 |
|------|------|------|-------------|
| `setup()` | CLI 入口，流程编排 | Click options | 无 |
| `_interactive_wizard()` | 交互引导收集配置 | 用户输入 | config dict |
| `_generate_env(config, path)` | 写入 .env 文件 | config dict, Path | `.env` 文件 |
| `_init_data_dirs(db_path)` | 创建数据目录 | db_path str | `./data/` 目录 |
| `_init_database(db_path)` | 异步包装初始化数据库 | db_path str | SQLite DB + 表结构 |
| `_gitignore_add_env(path)` | 追加 `.env` 到 .gitignore | Path | `.gitignore` 更新 |
| `_check_environment()` | Python 版本 + 依赖检测 | 无 | bool / raise |
| `_test_llm_connection(settings)` | 发送测试请求验证 LLM | Settings | bool + message |
| `_print_summary(config)` | 打印配置摘要表格 | config dict | 无 |

class Settings:

        """将当前配置导出为 .env 文件"""
        lines = [
            f"YELLOWBULL_LLM_PROVIDER={self.llm.provider}",

            # ... 其他字段
        ]
        path.write_text("\n".join(lines), encoding="utf-8")

    @classmethod
    def from_env_file(cls, path: Path) -> Settings:
        """从 .env 文件加载配置"""
        return Settings(_env_file=str(path))
